
window._prospectToPRM = async function(idx) {
  var l = (window._lcAllLeads || [])[idx];
  if (!l) return;
  try {
    var nameParts = (l.contact || l.entreprise || 'Inconnu').split(' ');
    var firstName = nameParts[0] || 'Inconnu';
    var lastName = nameParts.slice(1).join(' ') || '';
    var notes = 'prm_stage: a_contacter\nsource_prm: leads_consolide\n'
      + (l.priorite ? 'priorite: '+l.priorite+'\n' : '')
      + (l.score ? 'score: '+l.score+'\n' : '')
      + (l.categorie ? 'categorie: '+l.categorie+'\n' : '')
      + (l.siret ? 'siret: '+l.siret+'\n' : '')
      + (l.site ? 'site_web: '+l.site+'\n' : '')
      + (l.notes ? 'notes_prospects: '+l.notes : '');
    var payload = {
      first_name: firstName,
      last_name: lastName,
      company: l.entreprise || '',
      phone: l.telephone || '',
      email: l.email || '',
      city: l.ville || '',
      source: 'prm',
      status: 'prospect',
      notes: notes
    };
    var res = await dbInsert('contacts', payload);
    if (res && res.id) {
      toast('✅ ' + (l.entreprise || l.contact || 'Prospect') + ' ajouté au PRM !', 'success');
      $('#deal-popup-overlay').style.display = 'none';
      navigate('prm');
    } else {
      toast('Erreur lors de l\'ajout au PRM', 'error');
    }
  } catch(e) {
    toast('Erreur : ' + e.message, 'error');
  }
};

window.dealView = async function(id) {
  var popup = $('#deal-popup-overlay');
  var popupBody = $('#deal-popup-body');
  popup.style.display = 'flex';
  popupBody.innerHTML = '<div class="text-center text-muted" style="padding:60px"><i class="fas fa-spinner fa-spin fa-2x"></i></div>';

  const [deals, contacts, activities, planningEvents] = await Promise.all([
    dbSelect('deals', { eq: [['id', id]] }),
    dbSelect('contacts'),
    dbSelect('activities', { order: ['created_at', false] }),
    dbSelect('planning', { order: ['start_date', true] })
  ]);

  const d = deals[0];
  if (!d) { popupBody.innerHTML = '<div class="empty-state"><i class="fas fa-exclamation-circle"></i><h4>Lead introuvable</h4></div>'; return; }

  const contactMap = {};
  contacts.forEach(function(c) { contactMap[c.id] = c; });
  const contact = contactMap[d.contact_id];
  const dealActivities = activities.filter(function(a) { return a.deal_id === d.id || a.contact_id === d.contact_id; });

  // Parse description for structured fields
  var parsedFields = parseDescription(d.notes || '');
  var stageColors = {
    nouveaux_leads: '#3498db', nrp: '#9b59b6', suivi: '#1abc9c',
    rdv_sur_place: '#2ecc71', devis: '#e0a030', attente_reponse: '#e67e22',
    r2: '#8e44ad', gagne: '#2d6a4f', perdu: '#e74c3c', foireux: '#78909c'
  };
  var stageLabels = {
    nouveaux_leads: 'Nouveaux leads', nrp: 'NRP', suivi: 'Suivi',
    rdv_sur_place: 'RDV sur place', devis: 'Devis', attente_reponse: 'Attente reponse',
    r2: 'R2', gagne: 'Gagne', perdu: 'Perdu', foireux: 'Leads foireux'
  };
  var stageColor = stageColors[d.stage] || '#888';
  var stageLabel = stageLabels[d.stage] || d.stage;
  var contactName = contact ? contact.first_name + ' ' + contact.last_name : d.title;
  var contactInits = contact ? initials(contact.first_name, contact.last_name) : (d.title ? d.title.replace(/[^A-Za-z\u00C0-\u024F ]/g, '').trim().split(/\s+/).map(function(w) { return w[0] || ''; }).join('').toUpperCase().substring(0, 2) : 'NN');

  // Extract phone & email from parsed fields or contact
  var phone = parsedFields.tel || parsedFields.telephone || parsedFields['tel.'] || parsedFields['tél'] || (contact && contact.phone) || '';
  var email = parsedFields.mail || parsedFields.email || (contact && contact.email) || '';
  var ville = parsedFields.commune || parsedFields.ville || parsedFields.city || (contact && contact.city) || '';

  // Extract attachments from notes (look for [FICHIER: ...] patterns)
  var attachments = extractAttachments(d.notes || '');

  // Build info fields from parsed description
  var infoFieldsHtml = '';
  var hiddenKeys = ['__raw'];
  var fieldsToShow = Object.entries(parsedFields).filter(function(kv) {
    if (hiddenKeys.indexOf(kv[0]) >= 0) return false;
    if (!kv[1] || !kv[1].trim()) return false;
    // Hide separator lines
    if (kv[1].trim() === '---') return false;
    return true;
  });
  fieldsToShow.forEach(function(kv) {
    var label = kv[0];
    var val = kv[1];
    // Format special fields
    var labelLC = label.toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g, '');
    if (['tel', 'telephone', 'tel.', 'tel:'].indexOf(labelLC) >= 0 || label.toLowerCase().match(/^t[eé]l/)) {
      val = '<a href="tel:' + val.replace(/\s/g, '') + '"><i class="fas fa-phone"></i> ' + escHtml(val) + '</a>';
    } else if (['mail', 'email'].indexOf(labelLC) >= 0) {
      val = '<a href="mailto:' + val + '"><i class="fas fa-envelope"></i> ' + escHtml(val) + '</a>';
    } else if (labelLC === 'source_url') {
      val = '<a href="' + escHtml(val) + '" target="_blank" rel="noopener" style="color:var(--primary)"><i class="fas fa-external-link-alt"></i> Voir le lead</a>';
      label = 'Lien source';
    } else {
      val = escHtml(val);
    }
    infoFieldsHtml += '<div class="deal-info-row"><div class="deal-info-label">' + escHtml(capitalize(label)) + '</div><div class="deal-info-value">' + val + '</div></div>';
  });

    return '<div class="deal-attachments" id="att-block">' +
      '<div class="att-header" style="display:flex;justify-content:space-between;align-items:center">' +
        '<span><i class="fas fa-paperclip"></i> Pièces jointes (' + atts.length + ')</span>' +
        '<button class="btn btn-outline btn-sm" id="btn-add-att" style="font-size:0.72rem"><i class="fas fa-plus"></i> Ajouter</button>' +
      '</div>' +
      (atts.length === 0 ? '<div class="text-muted text-small" style="padding:8px 4px">Aucune pièce jointe</div>' :
      '<div class="att-grid">' +
        atts.map(function(att, idx) {
          var iconClass = 'fas fa-file', colorClass = '';
          var nameLC = att.name.toLowerCase();
          var isImg = nameLC.match(/\.(jpg|jpeg|png|gif|heic|webp|bmp|svg)$/);
          if (isImg) { iconClass = 'fas fa-file-image'; colorClass = 'img'; }
          else if (nameLC.match(/\.(pdf)$/)) { iconClass = 'fas fa-file-pdf'; colorClass = 'pdf'; }
          else if (nameLC.match(/\.(doc|docx)$/)) { iconClass = 'fas fa-file-word'; colorClass = 'doc'; }
          else if (nameLC.match(/\.(xls|xlsx)$/)) { iconClass = 'fas fa-file-excel'; colorClass = 'doc'; }
          // Check local image cache
          var localImg = localStorage.getItem('att_img_' + d.id + '_' + idx);
          var urlToUse = att.url || localImg;
          var thumbHtml = (localImg && isImg) ? '<img src="' + localImg + '" style="width:40px;height:40px;object-fit:cover;border-radius:4px;margin-bottom:4px">' : '<div class="att-icon ' + colorClass + '"><i class="' + iconClass + '"></i></div>';
          return '<div class="att-item" style="position:relative" data-att-idx="' + idx + '">' +
            (urlToUse ? '<a href="' + escHtml(urlToUse) + '" target="_blank" style="text-decoration:none;color:inherit;display:contents">' : '') +
            thumbHtml +
            '<div class="att-name" title="' + escHtml(att.name) + '">' + escHtml(att.name) + '</div>' +
            (att.date ? '<div class="att-date">' + att.date + '</div>' : '') +
            (urlToUse ? '</a>' : '') +
            '<button class="btn-ghost att-del-btn" data-att-idx="' + idx + '" title="Supprimer" style="position:absolute;top:2px;right:2px;font-size:0.65rem;color:var(--danger);opacity:0;background:var(--card);border-radius:50%;width:18px;height:18px;padding:0;line-height:18px;text-align:center" onclick="event.preventDefault();event.stopPropagation()"><i class="fas fa-times"></i></button>' +

' (' + contactEvents.length + ')' : '') + '</span><button class="btn-ghost" onclick="planAddFromDeal(\'' + d.contact_id + '\',\'' + d.id + '\')" title="Ajouter un evenement"><i class="fas fa-plus"></i></button></div>' +

  var timelineHtml = '';

              '</span>' : '') +
            '</div>' +
            '<div class="tl-text" id="tl-text-' + actId + '">' + escHtml(item.text) + '</div>' +
          '</div>' +
        '</div>';
      }).join('') +
    '</div></div>';
  }

  $('#deal-popup-title').textContent = escHtml(d.title || contactName);
  popupBody.innerHTML = `
    <div class="deal-header" style="padding:0 0 12px 0;border-bottom:1px solid var(--border);margin-bottom:12px">
      <div class="deal-header-left">
        <div class="deal-avatar" style="background:${stageColor}">${contact ? avatarHtml(contact) : contactInits}</div>
        <div>
          <h2 style="font-size:1.2rem;margin:0">${escHtml(d.title || contactName)}</h2>
          <span class="deal-stage-badge" style="background:${stageColor}">${stageLabel}</span>
          ${ville ? ' <span style="font-size:0.82rem;color:var(--text-light);margin-left:8px"><i class="fas fa-map-marker-alt"></i> ' + escHtml(ville) + '</span>' : ''}
        </div>
      </div>
      <div class="deal-header-actions">
        ${phone ? '<a href="tel:' + phone.replace(/\s/g, '') + '" class="btn btn-outline btn-sm"><i class="fas fa-phone"></i> Appeler</a>' : ''}
        ${email ? '<button class="btn btn-outline btn-sm" onclick="window.emailCompose(\'' + email + '\',\'' + contactName.replace(/'/g,"\\'") + '\',\'' + d.id + '\',\'' + (d.contact_id||'') + '\')"><i class="fas fa-envelope"></i> Email</button>' : ''}
        <button class="btn btn-primary btn-sm" onclick="dealEditFromView('${d.id}')"><i class="fas fa-pen"></i> Modifier</button>
      </div>
    </div>

    <div class="deal-detail">
      <div class="deal-detail-main">
        ${infoFieldsHtml ? `
          <div class="deal-description-block">
            <div class="desc-header"><i class="fas fa-list-alt"></i> Informations du lead</div>
            <div class="deal-info-grid">${infoFieldsHtml}</div>
          </div>
        ` : ''}

        ${rawDesc ? `
          <div class="deal-description-block">
            <div class="desc-header"><i class="fas fa-align-left"></i> Description</div>
            <div class="desc-body">${escHtml(rawDesc)}</div>

        ${timelineHtml}

        <div class="quick-note-box" id="quick-note-box">
          <div class="qn-header"><i class="fas fa-pencil-alt"></i> Ajouter une note rapide</div>

    // Remove the [FICHIER: ...] tag at position idx from notes
    var count = -1;
    return (notes || '').replace(/\[FICHIER:[^\]]*\]/gi, function(match) {
      count++;
      return count === idx ? '' : match;
    }).replace(/\n{3,}/g, '\n\n').trim();
  }

    var tag = '[FICHIER: ' + name + (date ? ' - ' + date : '') + (url ? ' - ' + url : '') + ']';

  }

// Helper: escape HTML
function escHtml(str) {
  if (!str) return '';
  return String(str).replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;').replace(/"/g, '&quot;');
}

// Helper: capitalize first letter
function capitalize(str) {
  if (!str) return '';
  return str.charAt(0).toUpperCase() + str.slice(1);
}

// Parse description text into key-value fields
function parseDescription(text) {
  if (!text) return { __raw: '' };
  var result = {};
  var rawLines = [];
  var lines = text.split('\n');
  var inRawSection = false;

  for (var i = 0; i < lines.length; i++) {
    var line = lines[i];
    // Stop parsing key-value when we reach special sections
    if (line.match(/^---\s*(COMMENTAIRES|COMMENTS|HISTORIQUE|PIECES JOINTES|FICHIERS|ATTACHMENTS)/i)) {
      break;
    }
    // Bare "---" separator means switch to raw description
    if (line.trim() === '---') {
      inRawSection = true;
      continue;
    }
    if (inRawSection) {
      if (line.trim()) rawLines.push(line);
      continue;
    }

    // Try to parse key: value pairs (handle accented chars like tél, référence, etc.)
    var match = line.match(/^([A-Za-z\u00C0-\u024F\s._'-]+?)\s*:\s*(.*)$/);
    if (match && match[1].length < 40) {
      var key = match[1].trim();
      var val = (match[2] || '').trim();
      var keyLC = key.toLowerCase();
      // Skip keys that look like timestamps
      if (!keyLC.match(/^\d+\/\d+\/\d+/) && !keyLC.match(/^http/)) {
        // Only store if value is not empty
        if (val) {
          result[keyLC] = val;
        }
      } else {
        rawLines.push(line);
      }
    } else if (line.trim()) {
      rawLines.push(line);
    }
  }

  result.__raw = rawLines.join('\n').trim();
  return result;
}

// Extract attachments from notes text
function extractAttachments(text) {
  var attachments = [];
  // Pattern: [FICHIER: name.ext] or [FICHIER: name.ext - date]
  var regex = /\[FICHIER:\s*([^\]]+)\]/gi;
  var match;
  while ((match = regex.exec(text)) !== null) {
    var parts = match[1].split(' - ');
    attachments.push({
      name: parts[0].trim(),
      date: parts[1] ? parts[1].trim() : ''
    });
  }
  // Also look for lines like "- filename.ext (date) [url]" in attachment sections
  var inAttSection = false;
  text.split('\n').forEach(function(line) {
    if (line.match(/^---\s*(PIECES JOINTES|FICHIERS|ATTACHMENTS)/i) || line.match(/PIECES JOINTES/i)) {
      inAttSection = true;
      return;
    }
    if (inAttSection && line.match(/^---/)) { inAttSection = false; return; }
    if (inAttSection && line.trim()) {
      var m = line.match(/^[-*]\s*(.+?)\s*(?:\(([^)]+)\))?\s*(?:\[([^\]]+)\])?$/);
      if (m) {
        var name = m[1].trim();
        if (name && !attachments.find(function(a) { return a.name === name; })) {
          attachments.push({ name: name, date: m[2] ? m[2].trim() : '', url: m[3] ? m[3].trim() : '' });
        }
      }
    }
  });
  return attachments;
}

// Extract comments from notes text
function extractComments(text) {
  var comments = [];
  var inComments = false;
  var currentComment = null;
  var lines = text.split('\n');

  for (var i = 0; i < lines.length; i++) {
    var line = lines[i];
    if (line.match(/^---\s*(COMMENTAIRES|COMMENTS|HISTORIQUE)/i) || line.match(/^=+\s*(COMMENTAIRES|COMMENTS)/i)) {
      inComments = true;
      continue;
    }
    if (line.match(/^---\s*(PIECES JOINTES|FICHIERS)/i)) {
      inComments = false;
      continue;
    }
    if (!inComments) continue;

    // Detect comment header: "Name DD/MM/YYYY HH:MM" or similar
    var headerMatch = line.match(/^([A-Za-z\u00C0-\u024F][\w\u00C0-\u024F ]+?)\s+(\d{1,2}[\/.-]\d{1,2}[\/.-]\d{2,4}(?:\s+\d{1,2}[h:]\d{2})?)/);
    if (headerMatch) {
      if (currentComment) comments.push(currentComment);
      var dateStr = headerMatch[2].trim();
      var parsedDate = parseFlexDate(dateStr);
      currentComment = {
        author: headerMatch[1].trim(),
        date: dateStr,
        text: line.substring(headerMatch[0].length).replace(/^\s*[-:]\s*/, '').trim(),
        sortDate: parsedDate
      };
    } else if (currentComment && line.trim()) {
      currentComment.text += (currentComment.text ? '\n' : '') + line.trim();
    }
  }
  if (currentComment) comments.push(currentComment);
  return comments;
}

// Parse flexible date formats
function parseFlexDate(str) {
  if (!str) return new Date(0);
  var m = str.match(/(\d{1,2})[\/.-](\d{1,2})[\/.-](\d{2,4})(?:\s+(\d{1,2})[h:](\d{2}))?/);
  if (m) {
    var year = parseInt(m[3]);
    if (year < 100) year += 2000;
    return new Date(year, parseInt(m[2]) - 1, parseInt(m[1]), parseInt(m[4] || 0), parseInt(m[5] || 0));
  }
  return new Date(0);
}

    ]);

    // Search Console aggregates
    const scByQuery = {};
    scData.forEach(r => {
      if (!scByQuery[r.query]) scByQuery[r.query] = { clicks: 0, impressions: 0, positions: [] };
      scByQuery[r.query].clicks += r.clicks || 0;
      scByQuery[r.query].impressions += r.impressions || 0;
      scByQuery[r.query].positions.push(r.position || 0);
    });
    Object.values(scByQuery).forEach(q => {
      q.avgPosition = q.positions.length ? (q.positions.reduce((a, b) => a + b, 0) / q.positions.length).toFixed(1) : 0;
      q.ctr = q.impressions > 0 ? (q.clicks / q.impressions * 100).toFixed(2) : 0;

// SEO Keyword form
function seoKwForm(kw = null) {
  const isEdit = !!kw;
  const k = kw || {};
  openModal(isEdit ? 'Modifier le mot-cle' : 'Ajouter un mot-cle', `
    <div class="form-group"><label>Mot-cle *</label><input type="text" id="f-kword" value="${k.keyword || ''}" placeholder="Ex: maison bois ecologique"></div>
    <div class="form-row">
      <div class="form-group"><label>Position actuelle</label><input type="number" id="f-kpos" value="${k.position || ''}" min="1"></div>
      <div class="form-group"><label>Position precedente</label><input type="number" id="f-kprev" value="${k.previous_position || ''}" min="1"></div>
    </div>
    <div class="form-row">
      <div class="form-group"><label>Volume de recherche</label><input type="number" id="f-kvol" value="${k.search_volume || ''}"></div>
      <div class="form-group"><label>Difficulte (0-100)</label><input type="number" id="f-kdiff" value="${k.difficulty || ''}" min="0" max="100"></div>
    </div>
    <div class="form-group"><label>URL ciblee</label><input type="url" id="f-kurl" value="${k.target_url || ''}"></div>
    <div class="form-row">
      <div class="form-group">
        <label>Categorie</label>
        <select id="f-kcat">
          ${['','brand','produit','local','informationnel'].map(c => `<option value="${c}" ${k.category === c ? 'selected' : ''}>${c || '— Aucune —'}</option>`).join('')}
        </select>
      </div>
      <div class="form-group">
        <label>Statut</label>
        <select id="f-kstatus">
          ${['suivi','optimisation','atteint'].map(s => `<option value="${s}" ${k.status === s ? 'selected' : ''}>${s}</option>`).join('')}
        </select>
      </div>
    </div>
  `, `
    <button class="btn btn-outline" onclick="closeModal()">Annuler</button>
    <button class="btn btn-primary" id="btn-save-kw">${isEdit ? 'Mettre a jour' : 'Ajouter'}</button>
  `);
  $('#btn-save-kw').addEventListener('click', async () => {
    const keyword = $('#f-kword').value.trim();
    if (!keyword) { toast('Mot-cle requis', 'error'); return; }
    const row = {
      keyword,
      position: parseInt($('#f-kpos').value) || null,
      previous_position: parseInt($('#f-kprev').value) || null,
      search_volume: parseInt($('#f-kvol').value) || 0,
      difficulty: parseInt($('#f-kdiff').value) || 0,
      target_url: $('#f-kurl').value.trim() || null,
      category: $('#f-kcat').value || null,
      status: $('#f-kstatus').value,
      last_checked: new Date().toISOString()
    };
    if (isEdit) { await dbUpdate('seo_keywords', k.id, row); toast('Mot-cle mis a jour'); }
    else { await dbInsert('seo_keywords', row); toast('Mot-cle ajoute'); }
    closeModal();
    pages.seo();
  });
}

window.seoEditKw = async function(id) {
  const data = await dbSelect('seo_keywords', { eq: [['id', id]] });
  if (data[0]) seoKwForm(data[0]);
};
window.seoDeleteKw = async function(id) {
  await dbDelete('seo_keywords', id);
  toast('Mot-cle supprime');
  pages.seo();
};

// DASHBOARD DEVIS MODULE (style Toolcie)

pages['dashboard-devis'] = async function() {
  const content = $('#content');
  content.innerHTML = '<div class="text-center text-muted" style="padding:60px"><i class="fas fa-spinner fa-spin fa-2x"></i></div>';

  const [deals, contacts] = await Promise.all([
    dbSelect('deals', { order: ['created_at', false] }),
    dbSelect('contacts')
  ]);
  const contactMap = {};
  contacts.forEach(c => { contactMap[c.id] = c; });

  const statusColors = { envoye: '#f39c12', refuse: '#e74c3c', accepte: '#27ae60', attente: '#3498db', autres: '#95a5a6' };
  const statusLabels = { envoye: 'Envoyé (devis)', refuse: 'Refusé', accepte: 'Signé (client)', attente: 'En attente', autres: 'Autres étapes' };

  // Règles métier (mapping pipeline -> "devis")
  const val = d => parseFloat(d.value) || 0;
  const sumVal = arr => arr.reduce((s, d) => s + val(d), 0);

  const dealsPending = deals.filter(d => d.stage === 'attente_reponse'); // envoyés en attente
  const dealsRefused = deals.filter(d => d.stage === 'perdu');           // refusés (colonne "Perdu")
  const dealsWon     = deals.filter(d => d.stage === 'gagne');           // gagnés (colonne "Gagné")
  const signedContacts = contacts.filter(c => c.status === 'client');
  const signedContactIds = new Set(signedContacts.map(c => c.id));
  // deals de clients signés, mais pas encore comptés dans pending/refused/won
  const countedIds = new Set([...dealsPending, ...dealsRefused, ...dealsWon].map(d => d.id));
  const dealsSignedExtra = deals.filter(d => signedContactIds.has(d.contact_id) && !countedIds.has(d.id));
  const dealsSigned = [...dealsWon, ...dealsSignedExtra];

  // Envoyés = leads passés par "attente_reponse" → en attente + signés + refusés (sans doublon)
  const sentCount = dealsPending.length + dealsSigned.length + dealsRefused.length;
  const signedCount = signedContacts.length;
  const refusedCount = dealsRefused.length;
  const totalDevis = sentCount;

  const caSigned = sumVal(dealsSigned);   // Valeur du contrat des clients
  const caPending = sumVal(dealsPending); // opportunités en attente
  const caRefused = sumVal(dealsRefused);
  const signRate = sentCount > 0 ? (signedCount / sentCount * 100) : 0;
  const avgSigned = signedCount > 0 ? caSigned / signedCount : 0;

  // Evolution par mois (12 derniers mois) par statut
  const months = [];
  const now = new Date();
  const mKey = (y, mo) => y + '-' + String(mo + 1).padStart(2, '0');
  let yy = 2026, mm = 3; // avril 2026
  while (yy < now.getFullYear() || (yy === now.getFullYear() && mm <= now.getMonth())) {
    months.push({ key: mKey(yy, mm), label: new Date(yy, mm, 15).toLocaleDateString('fr-FR', { month: 'short', year: '2-digit' }) });
    mm++; if (mm > 11) { mm = 0; yy++; }
  }
  const bucketMonth = (dateStr) => (dateStr || '').slice(0, 7);
  const seriesByStatus = { envoye: [], refuse: [], accepte: [] };
  const monthSum = (arr, m) => arr.filter(d => bucketMonth(d.created_at) === m.key).reduce((s, d) => s + val(d), 0);
  months.forEach(m => {
    seriesByStatus.envoye.push(monthSum(dealsPending, m));
    seriesByStatus.refuse.push(monthSum(dealsRefused, m));
    seriesByStatus.accepte.push(monthSum(dealsSigned, m));
  });
  const totalByStatus = { envoye: caPending, refuse: caRefused, accepte: caSigned };

  // Répartition (count par catégorie "devis")
  const statusKeys = ['envoye', 'accepte', 'refuse'];
  const countByStatus = [dealsPending.length, signedCount, refusedCount];

  // Suivi client — prix des devis signés (par client signé)
  const clientRows = signedContacts.map(c => {
    const dls = deals.filter(d => d.contact_id === c.id);
    return { contact: c, total: dls.reduce((s, d) => s + val(d), 0), count: dls.length, devis: dls };
  }).filter(r => r.count > 0).sort((a, b) => b.total - a.total);

  const kpiCard = (bg, icon, value, label) =>
    '<div class="stat-card" style="background:' + bg + ';color:#fff;border:none">' +
      '<div style="display:flex;align-items:center;justify-content:space-between">' +
        '<i class="fas ' + icon + '" style="font-size:2.2rem;opacity:.85"></i>' +
        '<div style="text-align:right">' +
          '<div style="font-size:1.7rem;font-weight:700;line-height:1.1">' + value + '</div>' +
          '<div style="opacity:.9;font-size:.85rem">' + label + '</div>' +
        '</div>' +
      '</div>' +
    '</div>';

  content.innerHTML = `
    <div class="stats-grid" style="grid-template-columns:repeat(4,1fr);gap:14px;margin-bottom:16px">
      ${kpiCard('#e67e22', 'fa-file-invoice-dollar', totalDevis, 'Devis total')}
      ${kpiCard('#f39c12', 'fa-paper-plane', dealsPending.length, 'Envoyés')}
      ${kpiCard('#27ae60', 'fa-file-signature', signedCount, 'Signés')}
      ${kpiCard('#e74c3c', 'fa-times-circle', refusedCount, 'Refusés')}
    </div>
    <div class="stats-grid" style="grid-template-columns:repeat(4,1fr);gap:14px;margin-bottom:20px">
      ${kpiCard('#16a085', 'fa-euro-sign', formatCurrency(caSigned), 'CA signé')}
      ${kpiCard('#3498db', 'fa-hourglass-half', formatCurrency(caPending), 'CA en attente')}
      ${kpiCard('#5b6ebf', 'fa-percentage', signRate.toFixed(1) + ' %', 'Taux de signature')}
      ${kpiCard('#9b59b6', 'fa-shopping-cart', formatCurrency(avgSigned), 'Panier moyen signé')}
    </div>

    <div style="display:grid;grid-template-columns:1.6fr 1fr;gap:16px;margin-bottom:20px">
      <div class="panel">
        <div class="panel-header"><h3>Évolution des devis</h3><span class="text-muted text-small">depuis avril 2026</span></div>
        <div class="panel-body"><div style="position:relative;height:280px"><canvas id="chart-devis-evolution"></canvas></div>
          <div style="display:flex;gap:24px;margin-top:14px;flex-wrap:wrap">
            <div><span style="display:inline-block;width:10px;height:10px;border-radius:50%;background:#f39c12;margin-right:6px"></span><strong>Envoyé</strong> · ${formatCurrency(totalByStatus.envoye)}</div>
            <div><span style="display:inline-block;width:10px;height:10px;border-radius:50%;background:#e74c3c;margin-right:6px"></span><strong>Refusé</strong> · ${formatCurrency(totalByStatus.refuse)}</div>
            <div><span style="display:inline-block;width:10px;height:10px;border-radius:50%;background:#27ae60;margin-right:6px"></span><strong>Accepté</strong> · ${formatCurrency(totalByStatus.accepte)}</div>
          </div>
        </div>
      </div>
      <div class="panel">
        <div class="panel-header"><h3>Répartition des devis</h3></div>
        <div class="panel-body"><div style="position:relative;height:280px"><canvas id="chart-devis-repartition"></canvas></div></div>
      </div>
    </div>

    <div class="panel" style="margin-bottom:20px">
      <div class="panel-header">
        <h3><i class="fas fa-euro-sign"></i> Chiffre d'affaires</h3>
        <span class="text-muted text-small">CA signé vs en attente — depuis avril 2026</span>
      </div>
      <div class="panel-body">
        <div style="display:grid;grid-template-columns:repeat(3,1fr);gap:16px;margin-bottom:18px">
          <div style="padding:14px;border-radius:8px;background:#e8f8f0;border-left:4px solid #27ae60">
            <div style="color:#7f8c8d;font-size:.8rem">CA signé (réalisé)</div>
            <div style="font-size:1.5rem;font-weight:700;color:#1b4332">${formatCurrency(caSigned)}</div>
          </div>
          <div style="padding:14px;border-radius:8px;background:#ebf5fb;border-left:4px solid #3498db">
            <div style="color:#7f8c8d;font-size:.8rem">CA en attente</div>
            <div style="font-size:1.5rem;font-weight:700;color:#1a4971">${formatCurrency(caPending)}</div>
          </div>
          <div style="padding:14px;border-radius:8px;background:#fdecea;border-left:4px solid #e74c3c">
            <div style="color:#7f8c8d;font-size:.8rem">CA perdu (refusé)</div>

      </div>

            </tr></thead>
            <tbody>
              ${clientRows.map(r => {
                const c = r.contact;

                return '<tr>' +

// ============================================
pages.suivi = async function() {
  var content = $('#content');
  content.innerHTML = '<div class="text-center text-muted" style="padding:60px"><i class="fas fa-spinner fa-spin fa-2x"></i></div>';

  var [contacts, deals] = await Promise.all([
    dbSelect('contacts', { order: ['created_at', false] }),
    dbSelect('deals', { order: ['created_at', false] })
  ]);

  var SUIVI_STAGES = [
    { key: 'nouveaux_clients',  label: 'Nouveaux clients',    color: '#3498db', icon: 'fa-user-plus' },
    { key: 'devis_signe',       label: 'Devis signé',         color: '#2ecc71', icon: 'fa-file-signature' },
    { key: 'accompte_valide',   label: 'Acompte validé',      color: '#e0a030', icon: 'fa-euro-sign' },
    { key: 'chantier',          label: 'Chantier',            color: '#2d6a4f', icon: 'fa-hard-hat' },
    { key: 'termine_paye',      label: 'Terminé / Payé',      color: '#00897b', icon: 'fa-check-double' },
    { key: 'avis_google',       label: 'Avis Google validé',  color: '#f4a400', icon: 'fa-star' }
  ];

  // Helper: get/set suivi_stage from contact notes
  function getSuiviStage(c) {
    var m = (c.notes || '').match(/^suivi_stage:\s*(\S+)/m);
    return m ? m[1] : 'nouveaux_clients';
  }

  // Deals par contact
  var dealsByContact = {};
  deals.forEach(function(d) {
    if (!d.contact_id) return;
    if (!dealsByContact[d.contact_id]) dealsByContact[d.contact_id] = [];
    dealsByContact[d.contact_id].push(d);
  });

  // Clients = contacts avec suivi_stage OU deal gagné OU status client
  var clients = contacts.filter(function(c) {
    var hasStage = /^suivi_stage:/m.test(c.notes || '');
    var hasWon = (dealsByContact[c.id] || []).some(function(d){ return d.stage === 'gagne'; });
    return hasStage || hasWon || c.status === 'client';
  });

  // Source colors
  var SRC_COLORS = { 'travaux.com':'#2e7d32','helloartisan':'#1976d2','website':'#c62828','facebook':'#e65100' };

  function renderKanban() {
    var byStage = {};
    SUIVI_STAGES.forEach(function(s){ byStage[s.key] = []; });
    clients.forEach(function(c){
      var st = getSuiviStage(c);
      if (!byStage[st]) st = 'nouveaux_clients';
      byStage[st].push(c);
    });

    content.innerHTML = `
      <div class="toolbar" style="margin-bottom:12px">
        <div class="toolbar-left">
          <span class="text-muted"><i class="fas fa-handshake"></i> <strong>${clients.length}</strong> client(s) en suivi</span>
        </div>
        <div class="toolbar-right">
          <button class="btn btn-primary" id="btn-add-suivi-client"><i class="fas fa-plus"></i> Ajouter un client</button>
        </div>
      </div>
      <div class="pipeline" style="display:flex;gap:12px;overflow-x:auto;padding-bottom:16px;align-items:flex-start">
        ${SUIVI_STAGES.map(function(st) {
          var cards = byStage[st.key] || [];
          return '<div class="pipeline-col" style="min-width:220px;flex:0 0 220px">' +
            '<div class="pipeline-col-header" style="background:'+st.color+';color:#fff;padding:10px 14px;border-radius:10px 10px 0 0;display:flex;align-items:center;gap:8px">' +
              '<i class="fas '+st.icon+'"></i>' +
              '<span style="font-weight:700;font-size:0.85rem;flex:1">'+escHtml(st.label)+'</span>' +
              '<span style="background:rgba(255,255,255,0.25);border-radius:12px;padding:1px 8px;font-size:0.75rem;font-weight:700">'+cards.length+'</span>' +
            '</div>' +
            '<div class="suivi-col-body" data-suivi-stage="'+st.key+'" style="background:var(--bg);border:1px solid var(--border);border-top:none;border-radius:0 0 10px 10px;padding:10px;min-height:200px;display:flex;flex-direction:column;gap:8px">' +
              cards.map(function(c) {
                var cDeals = dealsByContact[c.id] || [];
                var bestDeal = cDeals.sort(function(a,b){ return (parseFloat(b.value)||0)-(parseFloat(a.value)||0); })[0];
                var srcCol = SRC_COLORS[c.source] || '#78909c';
                return '<div class="suivi-card" draggable="true" data-contact-id="'+c.id+'" onclick="window.suiviView(\''+c.id+'\')" style="background:var(--card);border-radius:8px;padding:12px;box-shadow:0 1px 4px rgba(0,0,0,0.07);cursor:pointer;border-left:3px solid '+srcCol+';transition:box-shadow 0.15s" onmouseover="this.style.boxShadow=\'0 4px 12px rgba(0,0,0,0.12)\'" onmouseout="this.style.boxShadow=\'0 1px 4px rgba(0,0,0,0.07)\'">' +
                  '<div style="display:flex;align-items:center;gap:8px;margin-bottom:6px">' +
                    '<div class="avatar-sm" style="width:30px;height:30px;font-size:0.65rem;flex-shrink:0">'+avatarHtml(c)+'</div>' +
                    '<div style="min-width:0">' +
                      '<div style="font-weight:700;font-size:0.85rem;white-space:nowrap;overflow:hidden;text-overflow:ellipsis">'+escHtml(c.first_name+' '+c.last_name)+'</div>' +
                      (c.company ? '<div style="font-size:0.72rem;color:var(--text-light);white-space:nowrap;overflow:hidden;text-overflow:ellipsis">'+escHtml(c.company)+'</div>' : '') +
                    '</div>' +
                  '</div>' +
                  (c.phone ? '<div style="font-size:0.75rem;color:var(--text-light);margin-bottom:3px"><i class="fas fa-phone" style="width:12px"></i> '+escHtml(c.phone)+'</div>' : '') +
                  (c.city ? '<div style="font-size:0.75rem;color:var(--text-light);margin-bottom:3px"><i class="fas fa-map-marker-alt" style="width:12px"></i> '+escHtml(c.city)+'</div>' : '') +
                  (bestDeal && bestDeal.value > 0 ? '<div style="font-size:0.82rem;font-weight:700;color:var(--primary);margin-top:4px">'+formatCurrency(bestDeal.value)+'</div>' : '') +
                  '<div style="display:flex;justify-content:flex-end;margin-top:6px;gap:4px" onclick="event.stopPropagation()">' +
                    SUIVI_STAGES.filter(function(s){ return s.key !== st.key; }).map(function(s2) {
                      return '<button title="→ '+s2.label+'" onclick="window._suiviMove(\''+c.id+'\',\''+s2.key+'\')" style="background:'+s2.color+'22;border:1px solid '+s2.color+'44;color:'+s2.color+';border-radius:4px;padding:1px 5px;font-size:0.6rem;cursor:pointer;white-space:nowrap">'+s2.label.split(' ')[0]+'→</button>';
                    }).join('') +
                  '</div>' +
                '</div>';
              }).join('') +
            '</div>' +
          '</div>';
        }).join('')}
      </div>
    `;

    // Drag & drop
    content.querySelectorAll('.suivi-card').forEach(function(card) {
      card.addEventListener('dragstart', function(e) {
        e.dataTransfer.setData('text/plain', card.dataset.contactId);
        card.style.opacity = '0.4';
      });
      card.addEventListener('dragend', function() { card.style.opacity = '1'; });
    });
    content.querySelectorAll('.suivi-col-body').forEach(function(col) {
      col.addEventListener('dragover', function(e) { e.preventDefault(); col.style.background = 'var(--primary-light)'; });
      col.addEventListener('dragleave', function() { col.style.background = 'var(--bg)'; });
      col.addEventListener('drop', async function(e) {
        e.preventDefault();
        col.style.background = 'var(--bg)';
        var cid = e.dataTransfer.getData('text/plain');
        await window._suiviMove(cid, col.dataset.suiviStage);
      });
    });

    // Ajouter client
    content.querySelector('#btn-add-suivi-client')?.addEventListener('click', function() {
      var available = contacts.filter(function(c){ return !clients.find(function(cl){ return cl.id===c.id; }); });
      openModal('Ajouter un client au suivi', `
        <div class="form-group">
          <label>Client</label>
          <select id="f-suivi-contact" style="width:100%;padding:8px;border:1px solid var(--border);border-radius:6px">
            <option value="">— Choisir —</option>
            ${contacts.map(function(c){ return '<option value="'+c.id+'">'+escHtml(c.first_name+' '+c.last_name+(c.company?' ('+c.company+')':''))+'</option>'; }).join('')}
          </select>
        </div>
        <div class="form-group">
          <label>Étape initiale</label>
          <select id="f-suivi-stage" style="width:100%;padding:8px;border:1px solid var(--border);border-radius:6px">
            ${SUIVI_STAGES.map(function(s){ return '<option value="'+s.key+'">'+s.label+'</option>'; }).join('')}
          </select>
        </div>
      `, `
        <button class="btn btn-outline" onclick="closeModal()">Annuler</button>
        <button class="btn btn-primary" id="btn-confirm-suivi">Ajouter</button>
      `);
      $('#btn-confirm-suivi').addEventListener('click', async function() {
        var cid = $('#f-suivi-contact').value;
        var stage = $('#f-suivi-stage').value;
        if (!cid) { toast('Choisissez un client', 'warning'); return; }
        var c = contacts.find(function(x){ return x.id===cid; });
        if (!c) return;
        var newNotes = (c.notes || '').replace(/^suivi_stage:.*$/m, '').trim();
        newNotes = 'suivi_stage: '+stage+'\n'+(newNotes||'');
        await dbUpdate('contacts', cid, { notes: newNotes.trim(), status: 'client' });
        c.notes = newNotes.trim(); c.status = 'client';
        if (!clients.find(function(x){ return x.id===cid; })) clients.push(c);
        closeModal();
        renderKanban();
        toast(escHtml(c.first_name+' '+c.last_name)+' ajouté au suivi ✓');
      });
    });
  }

  window._suiviMove = async function(contactId, newStage) {
    var c = clients.find(function(x){ return x.id===contactId; });
    if (!c) return;
    var newNotes = (c.notes || '').replace(/^suivi_stage:.*$/m, '').trim();
    newNotes = 'suivi_stage: '+newStage+'\n'+(newNotes||'');
    await dbUpdate('contacts', contactId, { notes: newNotes.trim() });
    c.notes = newNotes.trim();
    renderKanban();
    var stageName = (SUIVI_STAGES.find(function(s){ return s.key===newStage; })||{}).label||newStage;
    toast('Déplacé vers "'+stageName+'" ✓');
  };

  renderKanban();
};

          </div>
        </div>

        <!-- Date de chantier -->
        <div class="deal-sidebar-card">
          <div class="sc-header"><i class="fas fa-hard-hat"></i> Date de chantier</div>
          <div class="sc-body" style="padding:12px 16px">
            <input type="date" id="f-chantier-date" value="${(function(){ var m=(c.notes||'').match(/^chantier_date:\s*(\S+)/m); return m?m[1]:''; })()}" style="width:100%;border:1px solid var(--border);border-radius:8px;padding:8px 10px;font-size:0.9rem;background:var(--bg);color:var(--text);cursor:pointer">
            <div id="chantier-date-display" style="margin-top:8px;font-size:0.8rem;color:var(--text-light);text-align:center"></div>
          </div>
        </div>

        <!-- Lien vers fiche CRM -->
        ${bestDeal ? `<div class="deal-sidebar-card" style="text-align:center;padding:12px">
          <button onclick="closeDealPopup();dealView('${bestDeal.id}')" class="btn btn-outline btn-sm" style="width:100%"><i class="fas fa-external-link-alt"></i> Voir fiche CRM</button>
        </div>` : ''}
      </div>
    </div>
  `;

  // Changement d'étape
  var stageSelect = popupBody.querySelector('#f-suivi-stage-select');
  stageSelect.addEventListener('change', async function() {
    var newStage = stageSelect.value;
    var sc = SUIVI_STAGES.find(function(s){ return s.key===newStage; }) || SUIVI_STAGES[0];
    stageSelect.style.background = sc.color;
    stageSelect.style.borderColor = sc.color;
    var newNotes = (c.notes || '').replace(/^suivi_stage:.*$/m, '').trim();
    newNotes = 'suivi_stage: '+newStage+'\n'+(newNotes||'');
    await dbUpdate('contacts', c.id, { notes: newNotes.trim() });
    c.notes = newNotes.trim();
    toast('Étape client → ' + sc.label + ' ✓');
  });

  // Date de chantier — save + sync planning
  var chantierInput = popupBody.querySelector('#f-chantier-date');
  var chantierDisplay = popupBody.querySelector('#chantier-date-display');

  function updateChantierDisplay(val) {
    if (!val) { chantierDisplay.textContent = ''; return; }
    var d = new Date(val + 'T00:00:00');
    var diff = Math.round((d - new Date()) / 86400000);
    var label = d.toLocaleDateString('fr-FR', { weekday:'long', day:'numeric', month:'long', year:'numeric' });
    var diffStr = diff === 0 ? "Aujourd'hui" : diff > 0 ? 'Dans ' + diff + ' jour' + (diff>1?'s':'') : Math.abs(diff) + ' jour' + (Math.abs(diff)>1?'s':'') + ' passé' + (Math.abs(diff)>1?'s':'');
    chantierDisplay.innerHTML = '<i class="fas fa-calendar-day"></i> ' + label + '<br><span style="color:' + (diff < 0 ? '#e74c3c' : diff <= 7 ? '#e0a030' : 'var(--primary)') + ';font-weight:700">' + diffStr + '</span>';
  }
  updateChantierDisplay(chantierInput.value);

  chantierInput.addEventListener('change', async function() {
    var val = chantierInput.value;
    // Sauvegarder dans notes contact
    var newNotes = (c.notes || '').replace(/^chantier_date:.*$/m, '').trim();
    newNotes = (newNotes || '') + '\nchantier_date: ' + (val || '');
    await dbUpdate('contacts', c.id, { notes: newNotes.trim() });
    c.notes = newNotes.trim();
    updateChantierDisplay(val);

    if (!val) { toast('Date de chantier effacée'); return; }

    // Créer/MAJ événement dans le planning
    var existingEvents = await dbSelect('planning', { eq: [['contact_id', c.id]] });
    var chantierEv = existingEvents.find(function(e){ return e.type === 'chantier'; });
    var evData = {
      title: 'Chantier — ' + c.first_name + ' ' + c.last_name,
      type: 'chantier',
      start_date: val,
      color: '#2d6a4f',
      contact_id: c.id,
      status: 'planifie',
      notes: (c.city ? c.city + ' ' : '') + (c.postal_code || '')
    };
    if (chantierEv) {
      await dbUpdate('planning', chantierEv.id, evData);
      // Sync GCal si connecté
      if (gcalIsConnected()) {
        var gcalEv = planningToGcalEvent(Object.assign({}, evData, { id: chantierEv.id, gcal_event_id: chantierEv.gcal_event_id }), c);
        var calId = getGcalCalendarId();
        if (chantierEv.gcal_event_id) await gcalApi('PUT', '/calendars/'+encodeURIComponent(calId)+'/events/'+chantierEv.gcal_event_id, gcalEv);
        else { var cr = await gcalApi('POST', '/calendars/'+encodeURIComponent(calId)+'/events', gcalEv); if(cr&&cr.id) await dbUpdate('planning', chantierEv.id, {gcal_event_id:cr.id}); }
      }
      toast('📅 Date de chantier mise à jour dans le planning ✓');
    } else {
      var inserted = await dbInsert('planning', evData);
      if (gcalIsConnected() && inserted) {
        var gcalEv2 = planningToGcalEvent(inserted, c);
        var calId2 = getGcalCalendarId();
        var cr2 = await gcalApi('POST', '/calendars/'+encodeURIComponent(calId2)+'/events', gcalEv2);
        if (cr2 && cr2.id) await dbUpdate('planning', inserted.id, { gcal_event_id: cr2.id });
      }
      toast('📅 Chantier ajouté au planning' + (gcalIsConnected() ? ' + Google Agenda ✓' : ' ✓'));
    }
  });

  // Note rapide
  var btnSaveNote = popupBody.querySelector('#btn-suivi-save-note');
  btnSaveNote.addEventListener('click', async function() {
    var text = popupBody.querySelector('#suivi-quick-note').value.trim();
    var type = popupBody.querySelector('#suivi-note-type').value;
    if (!text) return;
    await dbInsert('activities', { contact_id: c.id, deal_id: bestDeal ? bestDeal.id : null, type: type, notes: text });
    toast('Note enregistrée ✓');
    window.suiviView(c.id);
  });
};

    <div class="form-group">

    </div>

      numero: num,
      objet: $('#f-cmdobjet').value.trim() || null,

var planningView = '2weeks'; // 'week', '2weeks', 'month'
var planningOffset = 0; // offset in periods from today

pages.planning = async function() {
  var content = $('#content');
  content.innerHTML = '<div class="text-center text-muted" style="padding:60px"><i class="fas fa-spinner fa-spin fa-2x"></i></div>';

  var [events, contacts, commandes] = await Promise.all([
    dbSelect('planning', { order: ['start_date', true] }),
    dbSelect('contacts'),
    dbSelect('commandes')
  ]);
  var contactMap = {};
  contacts.forEach(function(c) { contactMap[c.id] = c; });

  // Cache global pour planningClientView (bannière rappels du jour)
  window._planContactMapCache = contactMap;
  window._planEvCache = {};
  events.forEach(function(ev){ window._planEvCache[ev.id] = ev; });

  // Compute date range based on view
  var now = new Date();
  var rangeStart, rangeEnd, headerLabel;
  var monthNames = ['Janvier','Fevrier','Mars','Avril','Mai','Juin','Juillet','Aout','Septembre','Octobre','Novembre','Decembre'];

  if (planningView === 'month') {
    var m = new Date(now.getFullYear(), now.getMonth() + planningOffset, 1);
    rangeStart = new Date(m);
    rangeEnd = new Date(m.getFullYear(), m.getMonth() + 1, 0);
    headerLabel = monthNames[m.getMonth()] + ' ' + m.getFullYear();
    // Align to Monday
    var dow = rangeStart.getDay() || 7;
    rangeStart.setDate(rangeStart.getDate() - (dow - 1));
    var endDow = rangeEnd.getDay() || 7;
    if (endDow < 7) rangeEnd.setDate(rangeEnd.getDate() + (7 - endDow));
  } else {
    var daysCount = planningView === 'week' ? 7 : 14;
    var today = new Date(now.getFullYear(), now.getMonth(), now.getDate());
    var todayDow = today.getDay() || 7;
    rangeStart = new Date(today);
    rangeStart.setDate(today.getDate() - (todayDow - 1) + planningOffset * daysCount);
    rangeEnd = new Date(rangeStart);
    rangeEnd.setDate(rangeStart.getDate() + daysCount - 1);
    var opts = { day: 'numeric', month: 'short' };
    headerLabel = rangeStart.toLocaleDateString('fr-FR', opts) + ' — ' + rangeEnd.toLocaleDateString('fr-FR', opts) + ' ' + rangeEnd.getFullYear();
  }

  // Group events by date — use lead name as title when available
  var eventsByDate = {};
  events.forEach(function(ev) {
    var d = ev.start_date;
    if (!d) return;
    var key = d.substring(0, 10);
    if (!eventsByDate[key]) eventsByDate[key] = [];
    var c = contactMap[ev.contact_id];
    var displayTitle = c ? c.first_name + ' ' + c.last_name : ev.title;
    eventsByDate[key].push(Object.assign({}, ev, { displayTitle: displayTitle }));
  });

  commandes.forEach(function(cmd) {
    if (!cmd.date_livraison) return;
    var key = cmd.date_livraison.substring(0, 10);
    if (!eventsByDate[key]) eventsByDate[key] = [];
    eventsByDate[key].push({ title: 'Livraison: ' + (cmd.objet || cmd.numero), displayTitle: 'Livraison: ' + (cmd.objet || cmd.numero), type: 'livraison', color: '#9b59b6', id: 'cmd-' + cmd.id });
  });

  // Calcul du nombre de lignes (semaines) dans la vue
  var totalDays = Math.round((rangeEnd - rangeStart) / 86400000) + 1;
  var numRows = Math.ceil(totalDays / 7);

  // Build calendar cells for range
  var calendarHtml = '';
  var cursor = new Date(rangeStart);
  while (cursor <= rangeEnd) {
    var dateKey = cursor.getFullYear() + '-' + String(cursor.getMonth() + 1).padStart(2, '0') + '-' + String(cursor.getDate()).padStart(2, '0');
    var dayEvents = eventsByDate[dateKey] || [];
    var isToday = cursor.getDate() === now.getDate() && cursor.getMonth() === now.getMonth() && cursor.getFullYear() === now.getFullYear();
    calendarHtml += '<div class="cal-cell' + (isToday ? ' cal-today' : '') + '" data-date="' + dateKey + '">' +
      '<div class="cal-day">' + cursor.getDate() + '</div>' +
      dayEvents.map(function(ev) {
        var timeStr = ev.start_time ? '<span style="font-weight:600">' + ev.start_time.substring(0,5) + '</span> ' : '';
        return '<div class="cal-event" data-evid="' + (ev.id || '') + '" style="background:' + (ev.color || '#2d6a4f') + ';cursor:pointer">' + timeStr + escHtml(ev.displayTitle) + '</div>';
      }).join('') +
    '</div>';
    cursor.setDate(cursor.getDate() + 1);
  }

  var typeLabels = { tache: 'Tache', rdv: 'RDV', chantier: 'Chantier', livraison: 'Livraison', autre: 'Autre' };

  // Rappels du jour
  var todayKey = now.getFullYear() + '-' + String(now.getMonth()+1).padStart(2,'0') + '-' + String(now.getDate()).padStart(2,'0');
  var todayEvents = events.filter(function(ev){ return ev.start_date && ev.start_date.substring(0,10) === todayKey && ev.status !== 'termine' && ev.status !== 'annule'; });
  todayEvents.sort(function(a,b){ return (a.start_time||'').localeCompare(b.start_time||''); });

  var typeIcons = { tache: 'fa-check-square', rdv: 'fa-handshake', chantier: 'fa-hard-hat', livraison: 'fa-truck', rappel: 'fa-bell', autre: 'fa-calendar-day' };
  var typeColors = { tache: '#3498db', rdv: '#2ecc71', chantier: '#2d6a4f', livraison: '#9b59b6', rappel: '#e67e22', autre: '#78909c' };

  var rappelsDuJourHtml = todayEvents.length === 0
    ? ''
    : '<div style="background:linear-gradient(135deg,#fff8e1,#fff3cd);border:1.5px solid #f4a400;border-radius:10px;padding:14px 18px;margin-bottom:16px;box-shadow:0 2px 8px #f4a40022">' +
        '<div style="display:flex;align-items:center;gap:10px;margin-bottom:12px">' +
          '<i class="fas fa-bell" style="color:#f4a400;font-size:1.1rem"></i>' +
          '<strong style="font-size:1rem;color:#7a5c00">Rappels du jour</strong>' +
          '<span style="background:#f4a400;color:#fff;font-size:0.72rem;font-weight:800;padding:2px 9px;border-radius:10px">'+todayEvents.length+'</span>' +
          '<span style="font-size:0.8rem;color:#a07c00;margin-left:4px">'+new Date().toLocaleDateString('fr-FR',{weekday:'long',day:'numeric',month:'long'})+'</span>' +
        '</div>' +
        '<div style="display:flex;flex-wrap:wrap;gap:8px">' +
          todayEvents.map(function(ev){
            var c = contactMap[ev.contact_id];
            var displayTitle = c ? c.first_name + ' ' + c.last_name : ev.title;
            var icon = typeIcons[ev.type] || 'fa-calendar-day';
            var col = ev.color || typeColors[ev.type] || '#78909c';
            var timeStr = ev.start_time ? ev.start_time.substring(0,5) : '';
            var clickAction = ev.contact_id ? 'window.planningClientView(window._planEvCache && window._planEvCache["'+ev.id+'"],window._planContactMapCache)' : 'planEdit(\''+ev.id+'\')';
            return '<div style="background:#fff;border:1.5px solid '+col+';border-radius:8px;padding:8px 14px;display:flex;align-items:center;gap:10px;cursor:pointer;min-width:200px;flex:1;max-width:320px;box-shadow:0 1px 4px rgba(0,0,0,0.06)" onclick="'+clickAction+'">' +
              '<i class="fas '+icon+'" style="color:'+col+';font-size:1rem;flex-shrink:0"></i>' +
              '<div style="flex:1;min-width:0">' +
                '<div style="font-weight:700;font-size:0.9rem;overflow:hidden;text-overflow:ellipsis;white-space:nowrap">'+escHtml(displayTitle)+'</div>' +
                (timeStr ? '<div style="font-size:0.75rem;color:#666">⏰ '+timeStr+(ev.end_time?' → '+ev.end_time.substring(0,5):'')+'</div>' : '') +
                (c && c.phone ? '<div style="font-size:0.72rem;color:var(--primary);margin-top:2px"><i class="fas fa-phone" style="font-size:0.65rem"></i> '+escHtml(c.phone)+'</div>' : '') +
              '</div>' +
              (ev.contact_id ? '<span title="Fiche client" style="font-size:0.68rem;color:var(--primary);margin-right:2px"><i class="fas fa-user-circle"></i></span>' : '') +
              '<button onclick="event.stopPropagation();planDone(\''+ev.id+'\')" title="Marquer terminé" style="background:#27ae60;color:#fff;border:none;border-radius:6px;padding:4px 8px;font-size:0.72rem;cursor:pointer;flex-shrink:0"><i class="fas fa-check"></i></button>' +
            '</div>';
          }).join('') +
        '</div>' +
      '</div>';

  content.innerHTML = `
    ${rappelsDuJourHtml}
    <div class="toolbar" style="flex-wrap:wrap;gap:8px">
      <div class="toolbar-left" style="display:flex;align-items:center;gap:10px">
        <button class="btn btn-outline btn-sm" id="btn-plan-prev"><i class="fas fa-chevron-left"></i></button>
        <h3 style="margin:0;min-width:180px;text-align:center"><i class="fas fa-calendar-alt"></i> ${headerLabel}</h3>
        <button class="btn btn-outline btn-sm" id="btn-plan-next"><i class="fas fa-chevron-right"></i></button>
        <button class="btn btn-outline btn-sm" id="btn-plan-today" style="margin-left:6px">Aujourd'hui</button>
      </div>
      <div class="toolbar-right" style="display:flex;align-items:center;gap:6px;flex-wrap:wrap">
        <div style="display:flex;border:1px solid var(--border);border-radius:6px;overflow:hidden">
          <button class="btn btn-sm ${planningView === 'week' ? 'btn-primary' : 'btn-outline'}" data-planview="week" style="border-radius:0;border:none">Semaine</button>
          <button class="btn btn-sm ${planningView === '2weeks' ? 'btn-primary' : 'btn-outline'}" data-planview="2weeks" style="border-radius:0;border:none">2 semaines</button>
          <button class="btn btn-sm ${planningView === 'month' ? 'btn-primary' : 'btn-outline'}" data-planview="month" style="border-radius:0;border:none">Mois</button>
        </div>
        ${gcalIsConnected() ? `
          <button class="btn btn-outline btn-sm" id="btn-gcal-import" title="Importer depuis Google Agenda"><i class="fab fa-google"></i> Importer</button>
          <button class="btn btn-outline btn-sm" id="btn-gcal-sync" title="Synchroniser vers Google Agenda"><i class="fas fa-sync-alt"></i> Sync Google</button>
          <span class="text-small" style="color:var(--success);padding:0 8px"><i class="fas fa-check-circle"></i> Google Agenda</span>
        ` : `
          <button class="btn btn-outline btn-sm" id="btn-gcal-connect" title="Connecter Google Agenda"><i class="fab fa-google"></i> Connecter Google Agenda</button>
        `}
        <button class="btn btn-primary" id="btn-add-event"><i class="fas fa-plus"></i> Nouvel evenement</button>
      </div>
    </div>
    <div style="border:1px solid var(--border);border-radius:var(--radius);overflow:hidden;margin-bottom:24px;box-shadow:var(--shadow)">

      title: title,
      type: $('#f-evtype').value,
      color: $('#f-evcolor').value,

      toast('Evenement supprime');
      closeModal();
      pages.planning();
    });
  }
}

pages.analytics = async function() {
  const content = $('#content');
  content.innerHTML = '<div class="text-center text-muted" style="padding:60px"><i class="fas fa-spinner fa-spin fa-2x"></i></div>';
  const ga4 = await dbSelect('ga4_data', { order: ['date', false] });

  const byDate = {};
  ga4.forEach(r => {
    if (!byDate[r.date]) byDate[r.date] = { sessions: 0, users: 0, bounceRates: [], durations: [], events: 0 };
    byDate[r.date].sessions += r.sessions || 0;
    byDate[r.date].users += r.users || 0;
    byDate[r.date].events += r.events || 0;
    if (r.bounce_rate) byDate[r.date].bounceRates.push(parseFloat(r.bounce_rate));
    if (r.avg_session_duration) byDate[r.date].durations.push(parseFloat(r.avg_session_duration));
  });
  const dates = Object.keys(byDate).sort();
  const totalSessions = dates.reduce((s, d) => s + byDate[d].sessions, 0);
  const totalUsers = dates.reduce((s, d) => s + byDate[d].users, 0);
  const totalEvents = dates.reduce((s, d) => s + byDate[d].events, 0);
  const avgSessionsDay = dates.length ? Math.round(totalSessions / dates.length) : 0;

  const allBounceRates = ga4.filter(r => r.bounce_rate).map(r => parseFloat(r.bounce_rate));
  const avgBounceRate = allBounceRates.length ? (allBounceRates.reduce((a, b) => a + b, 0) / allBounceRates.length * 100).toFixed(1) : '—';
  const allDurations = ga4.filter(r => r.avg_session_duration).map(r => parseFloat(r.avg_session_duration));
  const avgDuration = allDurations.length ? Math.round(allDurations.reduce((a, b) => a + b, 0) / allDurations.length) : 0;
  const durationFormatted = avgDuration > 0 ? `${Math.floor(avgDuration / 60)}m${(avgDuration % 60).toString().padStart(2, '0')}s` : '—';

  const bySource = {};
  ga4.forEach(r => {
    const src = r.source || '(direct)';
    if (!bySource[src]) bySource[src] = { sessions: 0, users: 0, events: 0 };
    bySource[src].sessions += r.sessions || 0;
    bySource[src].users += r.users || 0;
    bySource[src].events += r.events || 0;
  });
  const sources = Object.entries(bySource).sort((a, b) => b[1].sessions - a[1].sessions);
  const srcColors = { '(direct)': '#3498db', 'google': '#e74c3c', 'm.facebook.com': '#3b5998', 'ig': '#c13584', 'facebook': '#3b5998', 'lovable.dev': '#8b5cf6', 'search.google.com': '#f39c12', 'direct': '#3498db' };

  content.innerHTML = `
    <div class="stats-grid" style="margin-bottom:20px">
      <div class="stat-card"><div class="stat-icon blue"><i class="fas fa-mouse-pointer"></i></div><div class="stat-info"><h3>${formatNumber(totalSessions)}</h3><p>Sessions (30j)</p></div></div>
      <div class="stat-card"><div class="stat-icon green"><i class="fas fa-users"></i></div><div class="stat-info"><h3>${formatNumber(totalUsers)}</h3><p>Utilisateurs (30j)</p></div></div>
      <div class="stat-card"><div class="stat-icon purple"><i class="fas fa-chart-bar"></i></div><div class="stat-info"><h3>${avgSessionsDay}</h3><p>Sessions / jour moy.</p></div></div>
      <div class="stat-card"><div class="stat-icon gold"><i class="fas fa-stopwatch"></i></div><div class="stat-info"><h3>${durationFormatted}</h3><p>Duree moy. session</p></div></div>
      <div class="stat-card"><div class="stat-icon red"><i class="fas fa-sign-out-alt"></i></div><div class="stat-info"><h3>${avgBounceRate}${avgBounceRate !== '—' ? '%' : ''}</h3><p>Taux de rebond</p></div></div>
      <div class="stat-card"><div class="stat-icon blue"><i class="fas fa-bolt"></i></div><div class="stat-info"><h3>${formatNumber(totalEvents)}</h3><p>Evenements (30j)</p></div></div>
    </div>
    <div class="grid-2">
      <div class="panel">
        <div class="panel-header"><h3><i class="fas fa-chart-line"></i> Sessions par jour</h3></div>
        <div class="panel-body"><div class="chart-container"><canvas id="chart-ga4-sessions"></canvas></div></div>
      </div>
      <div class="panel">
        <div class="panel-header"><h3><i class="fas fa-chart-pie"></i> Sources de trafic</h3></div>
        <div class="panel-body"><div class="chart-container"><canvas id="chart-ga4-sources"></canvas></div></div>
      </div>
    </div>
    <div class="panel mt-24">
      <div class="panel-header">
        <h3><i class="fas fa-list"></i> Detail par source</h3>
        <button class="btn btn-sm btn-outline" id="btn-refresh-ga4"><i class="fas fa-sync-alt"></i> Actualiser</button>
      </div>
      <div class="panel-body" style="padding:0">
        ${sources.length === 0 ? '<div class="empty-state"><i class="fas fa-chart-line"></i><h4>Aucune donnee Analytics</h4><p>Configurez Windsor.ai dans les Parametres pour synchroniser.</p></div>' : `
          <table class="data-table">
            <thead><tr><th>Source</th><th>Sessions</th><th>Utilisateurs</th><th>Evenements</th><th>% du trafic</th></tr></thead>
            <tbody>
              ${sources.map(([src, data]) => {
                const pct = totalSessions > 0 ? (data.sessions / totalSessions * 100).toFixed(1) : 0;
                return `<tr><td><strong>${src}</strong></td><td>${formatNumber(data.sessions)}</td><td>${formatNumber(data.users)}</td><td>${formatNumber(data.events)}</td><td><div class="progress-bar" style="width:100px;display:inline-block;vertical-align:middle"><div class="progress-bar-fill" style="width:${pct}%;background:${srcColors[src] || '#95a5a6'}"></div></div> <span class="text-small text-muted">${pct}%</span></td></tr>`;
              }).join('')}
            </tbody>
          </table>
        `}

pages.prm = async function(subTab) {
  subTab = subTab || 'pipeline';
  var content = $('#content');
  content.innerHTML = '<div class="text-center text-muted" style="padding:60px"><i class="fas fa-spinner fa-spin fa-2x"></i></div>';

  if (subTab === 'consolide') { return renderLeadsConsolides(content, 'prm'); }

  var contacts = await dbSelect('contacts', { order: ['created_at', false] });
  var prmContacts = contacts.filter(function(c){ return c.source === 'prm'; });

  var PRM_STAGES = [
    { key: 'a_contacter',   label: 'À contacter',   color: '#3498db', icon: 'fa-user-clock' },
    { key: 'contacte',      label: 'Contacté',       color: '#9b59b6', icon: 'fa-phone-volume' },
    { key: 'en_discussion', label: 'En discussion',  color: '#e0a030', icon: 'fa-comments' },
    { key: 'devis_envoye',  label: 'Devis envoyé',   color: '#2ecc71', icon: 'fa-file-invoice' },
    { key: 'partenaire',    label: 'Partenaire',     color: '#2d6a4f', icon: 'fa-handshake' },
    { key: 'perdu_prm',     label: 'Perdu',          color: '#e74c3c', icon: 'fa-times-circle' }
  ];

  function getPrmStage(c) {
    var m = (c.notes||'').match(/^prm_stage:\s*(\S+)/m);
    return m ? m[1] : 'a_contacter';
  }

  function getPrmField(c, key) {
    var m = (c.notes||'').match(new RegExp('^'+key+':\\s*(.+)$', 'm'));
    return m ? m[1].trim() : '';
  }

  function renderPRM() {
    var stageMap = {};
    PRM_STAGES.forEach(function(s){ stageMap[s.key] = []; });
    prmContacts.forEach(function(c){
      var st = getPrmStage(c);
      if (!stageMap[st]) st = 'a_contacter';
      stageMap[st].push(c);
    });

    content.innerHTML =
      '<div class="tab-bar" style="margin-bottom:16px;display:flex;gap:0;border-bottom:2px solid var(--border)">' +
        '<button class="tab-btn active" data-prm-tab="pipeline" style="padding:10px 20px;background:none;border:none;border-bottom:2px solid var(--primary);margin-bottom:-2px;font-weight:600;cursor:pointer;color:var(--primary)"><i class="fas fa-stream"></i> Pipeline PRM</button>' +
        '<button class="tab-btn" data-prm-tab="consolide" style="padding:10px 20px;background:none;border:none;border-bottom:2px solid transparent;margin-bottom:-2px;cursor:pointer;color:var(--text-light)"><i class="fas fa-layer-group"></i> Leads consolidés</button>' +
      '</div>' +
      '<div class="toolbar" style="margin-bottom:16px">' +
        '<div class="toolbar-left"><span class="text-muted"><i class="fas fa-briefcase"></i> <strong>' + prmContacts.length + '</strong> prospect(s) pro</span></div>' +
        '<div class="toolbar-right"><button class="btn btn-primary" id="btn-add-prm"><i class="fas fa-plus"></i> Nouveau prospect pro</button></div>' +
      '</div>' +
      '<div class="pipeline">' +
        PRM_STAGES.map(function(st){
          var cards = stageMap[st.key] || [];
          return '<div class="pipeline-col">' +
            '<div class="pipeline-col-header" style="border-bottom-color:' + st.color + '">' +
              '<span><i class="fas ' + st.icon + '" style="color:' + st.color + ';margin-right:6px"></i>' + st.label + '</span>' +
              '<span class="count">' + cards.length + '</span>' +
            '</div>' +
            '<div class="pipeline-col-body" data-prm-stage="' + st.key + '">' +
              (cards.length === 0 ? '<div class="text-center text-muted text-small" style="padding:20px">—</div>' : '') +
              cards.map(function(c){
                var prio = getPrmField(c, 'priorite');
                var score = getPrmField(c, 'score');
                var cat = getPrmField(c, 'categorie');
                var prioCol = prio==='A'?'#e53935':prio==='B'?'#fb8c00':'#43a047';
                return '<div class="deal-card" draggable="true" data-prm-id="' + c.id + '" onclick="window.prmView(\'' + c.id + '\')" style="position:relative;cursor:pointer">' +
                  '<div style="display:flex;align-items:center;gap:8px;margin-bottom:4px">' +
                    '<div class="avatar-sm" style="width:28px;height:28px;font-size:0.65rem">' + avatarHtml(c) + '</div>' +
                    '<div style="flex:1;min-width:0">' +
                      '<strong style="font-size:0.85rem;display:block;overflow:hidden;text-overflow:ellipsis;white-space:nowrap">' + escHtml((c.company || (c.first_name+' '+c.last_name)).trim()) + '</strong>' +
                      (c.city ? '<span style="display:block;font-size:0.7rem;color:var(--text-light)"><i class="fas fa-map-marker-alt" style="font-size:0.6rem"></i> ' + escHtml(c.city) + '</span>' : '') +
                    '</div>' +
                    (prio ? '<span style="background:' + prioCol + ';color:#fff;font-size:0.65rem;font-weight:800;padding:1px 6px;border-radius:3px">' + escHtml(prio) + '</span>' : '') +
                  '</div>' +
                  (cat ? '<div style="font-size:0.72rem;color:var(--text-light);margin-top:3px"><i class="fas fa-tag"></i> ' + escHtml(cat) + '</div>' : '') +
                  (score ? '<div style="font-size:0.78rem;font-weight:700;color:var(--primary);margin-top:4px">Score : ' + escHtml(score) + '</div>' : '') +
                  '<div style="display:flex;justify-content:space-between;align-items:center;font-size:0.7rem;color:var(--text-light);margin-top:6px">' +
                    (c.phone ? '<a href="tel:' + c.phone.replace(/\s/g,'') + '" onclick="event.stopPropagation()" style="color:var(--primary)"><i class="fas fa-phone"></i></a>' : '<span></span>') +
                    (c.email ? '<a href="mailto:' + escHtml(c.email) + '" onclick="event.stopPropagation()" style="color:var(--primary)"><i class="fas fa-envelope"></i></a>' : '<span></span>') +
                  '</div>' +
                '</div>';
              }).join('') +
            '</div>' +
            '<div class="text-center text-small text-muted" style="padding:6px;border-top:1px solid var(--border)">' + cards.length + ' prospect(s)</div>' +
          '</div>';
        }).join('') +
      '</div>';

    content.querySelector('#btn-add-prm')?.addEventListener('click', function(){ window.prmForm(); });

    // Tab navigation
    content.querySelectorAll('[data-prm-tab]').forEach(function(btn){
      btn.addEventListener('click', function(){ pages.prm(btn.dataset.prmTab); });
    });

    // Drag & Drop
    content.querySelectorAll('[data-prm-id]').forEach(function(card){
      card.addEventListener('dragstart', function(e){ e.dataTransfer.setData('text/plain', card.dataset.prmId); card.classList.add('dragging'); });
      card.addEventListener('dragend', function(){ card.classList.remove('dragging'); content.querySelectorAll('.pipeline-col-body').forEach(function(col){ col.classList.remove('drag-over'); }); });
    });
    content.querySelectorAll('[data-prm-stage]').forEach(function(col){
      col.addEventListener('dragover', function(e){ e.preventDefault(); });
      col.addEventListener('dragenter', function(e){ e.preventDefault(); col.classList.add('drag-over'); });
      col.addEventListener('dragleave', function(e){ if(!col.contains(e.relatedTarget)) col.classList.remove('drag-over'); });
      col.addEventListener('drop', async function(e){
        e.preventDefault(); col.classList.remove('drag-over');
        var cId = e.dataTransfer.getData('text/plain');
        var newStage = col.dataset.prmStage;
        if (!cId || !newStage) return;
        var c = prmContacts.find(function(x){ return x.id === cId; });
        if (!c) return;
        var newNotes = (c.notes||'').replace(/^prm_stage:\s*\S+/m, 'prm_stage: '+newStage);
        if (!(c.notes||'').match(/^prm_stage:/m)) newNotes = 'prm_stage: '+newStage+'\n'+(c.notes||'');
        await dbUpdate('contacts', cId, { notes: newNotes });
        c.notes = newNotes;
        toast('Déplacé vers ' + newStage.replace(/_/g,' '));
        renderPRM();
      });
    });
  }

  renderPRM();
};

      email: body.querySelector('#pf-email').value.trim(),
      city: body.querySelector('#pf-city').value.trim(),
      source: 'prm',
      status: 'prospect',
      notes: notes
    };
    var res = await dbInsert('contacts', payload);
    if (res && res.id) {
      toast('Prospect pro créé !', 'success');
      popup.style.display = 'none';
      pages.prm();
    } else {
      toast('Erreur lors de la création', 'error');
    }
  });
  popup.onclick = function(e){ if(e.target===popup) popup.style.display='none'; };
};

  toast('Prospect supprimé', 'success');
  $('#deal-popup-overlay').style.display = 'none';
  pages.prm();
};

pages.rapport = async function() {
  var content = $('#content');
  content.innerHTML = '<div class="text-center text-muted" style="padding:60px"><i class="fas fa-spinner fa-spin fa-2x"></i></div>';

  var df = getDateFilters('date');
  var dfCreated = getDateFilters('created_at');
  var dfStartDate = getDateFilters('start_date');

  var [contacts, deals, planning, activities, ga4, gmbData, adsData, scData] = await Promise.all([
    dbSelect('contacts', Object.assign({ order: ['created_at', false] }, dfCreated)),
    dbSelect('deals', Object.assign({ order: ['created_at', false] }, dfCreated)),
    dbSelect('planning', Object.assign({ order: ['start_date', false] }, dfStartDate)),
    dbSelect('activities', Object.assign({ order: ['created_at', false] }, dfCreated)),
    dbSelect('ga4_data', Object.assign({ order: ['date', false] }, df)),
    dbSelect('gmb_data', Object.assign({ order: ['date', false] }, df)),
    dbSelect('google_ads_data', Object.assign({ order: ['date', false] }, df)),
    dbSelect('search_console_data', Object.assign({ order: ['date', false] }, df))
  ]);

  // Group data by date for daily reports
  var dailyMap = {};

  function ensureDay(d) {
    if (!dailyMap[d]) dailyMap[d] = { newLeads: [], rdv: [], activities: [], ads: { impressions: 0, clicks: 0, spend: 0, conversions: 0 }, seo: { clicks: 0, impressions: 0, queries: [] }, ga4: { sessions: 0, users: 0 }, gmb: { impressions: 0, clicks: 0, directions: 0, calls: 0 }, deals: [] };
    return dailyMap[d];
  }

  contacts.forEach(function(c) {
    var d = (c.created_at || '').slice(0, 10);
    if (d) ensureDay(d).newLeads.push(c);
  });
  deals.forEach(function(deal) {
    var d = (deal.created_at || '').slice(0, 10);
    if (d) ensureDay(d).deals.push(deal);
  });
  planning.forEach(function(ev) {
    var d = (ev.start_date || '').slice(0, 10);
    if (d) ensureDay(d).rdv.push(ev);
  });
  activities.forEach(function(a) {
    var d = (a.created_at || '').slice(0, 10);
    if (d) ensureDay(d).activities.push(a);
  });
  adsData.forEach(function(r) {
    var d = r.date;
    if (d) { var day = ensureDay(d); day.ads.impressions += r.impressions || 0; day.ads.clicks += r.clicks || 0; day.ads.spend += parseFloat(r.spend) || 0; day.ads.conversions += r.conversions || 0; }
  });
  scData.forEach(function(r) {
    var d = r.date;
    if (d) { var day = ensureDay(d); day.seo.clicks += r.clicks || 0; day.seo.impressions += r.impressions || 0; if (r.query) day.seo.queries.push(r.query); }
  });
  ga4.forEach(function(r) {
    var d = r.date;
    if (d) { var day = ensureDay(d); day.ga4.sessions += r.sessions || 0; day.ga4.users += r.users || 0; }
  });
  gmbData.forEach(function(r) {
    var d = r.date;
    if (d) { var day = ensureDay(d); day.gmb.impressions += r.impressions || 0; day.gmb.clicks += r.clicks || 0; day.gmb.directions += r.direction_requests || 0; day.gmb.calls += r.call_clicks || 0; }
  });

  var sortedDays = Object.keys(dailyMap).sort().reverse();
  if (sortedDays.length === 0) {
    content.innerHTML = '<div class="empty-state"><i class="fas fa-clipboard-list"></i><h4>Aucun rapport pour cette periode</h4><p>Selectionnez une periode avec des donnees</p></div>';
    return;
  }

  // Summary KPIs for full period
  var totalNewLeads = contacts.filter(function(c) { return c.status === 'lead' || c.status === 'prospect'; }).length;
  var totalRdv = planning.length;
  var totalAdsSpend = adsData.reduce(function(s, r) { return s + (parseFloat(r.spend) || 0); }, 0);
  var totalSeoClicks = scData.reduce(function(s, r) { return s + (r.clicks || 0); }, 0);

  content.innerHTML = `
    <div class="stats-grid">
      <div class="stat-card">
        <div class="stat-icon green"><i class="fas fa-user-plus"></i></div>
        <div class="stat-info"><h3>${totalNewLeads}</h3><p>Nouveaux leads</p></div>
      </div>
      <div class="stat-card">
        <div class="stat-icon blue"><i class="fas fa-calendar-check"></i></div>
        <div class="stat-info"><h3>${totalRdv}</h3><p>RDV planifies</p></div>
      </div>
      <div class="stat-card">
        <div class="stat-icon gold"><i class="fas fa-ad"></i></div>
        <div class="stat-info"><h3>${formatCurrency(totalAdsSpend)}</h3><p>Budget Ads depense</p></div>
      </div>
      <div class="stat-card">
        <div class="stat-icon purple"><i class="fas fa-search"></i></div>
        <div class="stat-info"><h3>${totalSeoClicks}</h3><p>Clics SEO organiques</p></div>
      </div>
    </div>

    <div class="panel">
      <div class="panel-header"><h3><i class="fas fa-clipboard-list"></i> Rapports quotidiens</h3></div>
      <div class="panel-body" id="rapport-days" style="max-height:calc(100vh - 340px);overflow-y:auto"></div>
    </div>
  `;

  var rapportContainer = $('#rapport-days');
  rapportContainer.innerHTML = sortedDays.map(function(day) {
    var d = dailyMap[day];
    var dayLabel = new Date(day + 'T12:00:00').toLocaleDateString('fr-FR', { weekday: 'long', day: 'numeric', month: 'long', year: 'numeric' });

    var sections = [];

    // RDV
    if (d.rdv.length > 0) {
      sections.push('<div class="rapport-section"><h5><i class="fas fa-calendar-check" style="color:#3498db"></i> RDV (' + d.rdv.length + ')</h5><ul>' +
        d.rdv.map(function(ev) { return '<li><strong>' + (ev.title || 'Sans titre') + '</strong>' + (ev.start_time ? ' a ' + ev.start_time : '') + ' <span class="badge-sm ' + (ev.status === 'termine' ? 'bg-green' : 'bg-blue') + '">' + (ev.status || 'planifie') + '</span></li>'; }).join('') +
        '</ul></div>');
    }

    // Nouveaux leads
    if (d.newLeads.length > 0) {
      sections.push('<div class="rapport-section"><h5><i class="fas fa-user-plus" style="color:#2ecc71"></i> Nouveaux contacts (' + d.newLeads.length + ')</h5><ul>' +
        d.newLeads.map(function(c) { return '<li><strong>' + (c.first_name || '') + ' ' + (c.last_name || '') + '</strong>' + (c.company ? ' — ' + c.company : '') + (c.source ? ' <span class="badge-sm bg-gray">' + c.source + '</span>' : '') + '</li>'; }).join('') +
        '</ul></div>');
    }

    // Deals
    if (d.deals.length > 0) {
      var dealVal = d.deals.reduce(function(s, deal) { return s + (parseFloat(deal.value) || 0); }, 0);
      sections.push('<div class="rapport-section"><h5><i class="fas fa-hand-holding-usd" style="color:#e0a030"></i> Deals crees (' + d.deals.length + ' — ' + formatCurrency(dealVal) + ')</h5><ul>' +
        d.deals.map(function(deal) { return '<li><strong>' + (deal.title || 'Sans titre') + '</strong> — ' + formatCurrency(parseFloat(deal.value) || 0) + '</li>'; }).join('') +
        '</ul></div>');
    }

    // Google Ads
    if (d.ads.impressions > 0 || d.ads.spend > 0) {
      sections.push('<div class="rapport-section"><h5><i class="fas fa-ad" style="color:#e67e22"></i> Google Ads</h5>' +
        '<div class="rapport-metrics"><span><strong>' + d.ads.impressions + '</strong> impressions</span><span><strong>' + d.ads.clicks + '</strong> clics</span><span><strong>' + formatCurrency(d.ads.spend) + '</strong> depense</span><span><strong>' + d.ads.conversions + '</strong> conversions</span></div></div>');
    }

    // SEO (Search Console)
    if (d.seo.impressions > 0) {
      var topQueries = d.seo.queries.slice(0, 5);
      sections.push('<div class="rapport-section"><h5><i class="fas fa-search" style="color:#9b59b6"></i> SEO — Search Console</h5>' +
        '<div class="rapport-metrics"><span><strong>' + d.seo.impressions + '</strong> impressions</span><span><strong>' + d.seo.clicks + '</strong> clics</span></div>' +
        (topQueries.length > 0 ? '<div class="rapport-queries">Requetes : ' + topQueries.map(function(q) { return '<code>' + q + '</code>'; }).join(', ') + '</div>' : '') +
        '</div>');
    }

    // GA4
    if (d.ga4.sessions > 0) {
      sections.push('<div class="rapport-section"><h5><i class="fas fa-chart-line" style="color:#3498db"></i> Analytics (GA4)</h5>' +
        '<div class="rapport-metrics"><span><strong>' + d.ga4.sessions + '</strong> sessions</span><span><strong>' + d.ga4.users + '</strong> utilisateurs</span></div></div>');
    }

    // GMB
    if (d.gmb.impressions > 0) {
      sections.push('<div class="rapport-section"><h5><i class="fab fa-google" style="color:#4285f4"></i> Google Business Profile</h5>' +
        '<div class="rapport-metrics"><span><strong>' + d.gmb.impressions + '</strong> impressions</span><span><strong>' + d.gmb.clicks + '</strong> clics</span><span><strong>' + d.gmb.directions + '</strong> itineraires</span><span><strong>' + d.gmb.calls + '</strong> appels</span></div></div>');
    }

    if (sections.length === 0) return '';

    return '<div class="rapport-day">' +
      '<div class="rapport-day-header"><h4>' + dayLabel + '</h4></div>' +
      '<div class="rapport-day-body">' + sections.join('') + '</div>' +
      '</div>';
  }).filter(function(s) { return s; }).join('');
};

async function checkAuth() {
  if (!db) return false;
  const { data: { session } } = await db.auth.getSession();
  return !!session;
}

async function handleLogin() {
  const btn = $('#login-btn');
  const errEl = $('#login-error');
  const email = $('#login-email').value.trim();
  const password = $('#login-password').value;
  errEl.style.display = 'none';

  if (!email || !password) {
    errEl.textContent = 'Veuillez remplir tous les champs';
    errEl.style.display = 'block';
    return;
  }

  btn.disabled = true;
  btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Connexion...';

  const { error } = await db.auth.signInWithPassword({ email, password });

  if (error) {
    errEl.textContent = 'Email ou mot de passe incorrect';
    errEl.style.display = 'block';
    btn.disabled = false;
    btn.innerHTML = '<i class="fas fa-sign-in-alt"></i> Se connecter';
    return;
  }

  showApp();
}

async function handleLogout() {
  await db.auth.signOut();
  $('#login-overlay').classList.remove('hidden');
  document.querySelector('.sidebar').style.display = 'none';
  document.querySelector('.main').style.display = 'none';
}

function showApp() {
  $('#login-overlay').classList.add('hidden');
  document.querySelector('.sidebar').style.display = '';
  document.querySelector('.main').style.display = '';
  navigate('dashboard');

  // Auto-sync Google Calendar at login
  // Tente un token silencieux si Client ID configuré, sinon utilise le token existant
  setTimeout(async function() {
    var clientId = getGcalClientId();
    if (!clientId) return; // GCal pas configuré

    async function doSync() {
      try {
        var synced = await gcalSyncAll();
        var msg = [];
        if (synced && synced.created > 0) msg.push(synced.created + ' sync(s) vers GCal');
        if (synced && synced.updated > 0) msg.push(synced.updated + ' mis à jour');
        if (msg.length > 0) toast('📅 Planning synchronisé : ' + msg.join(', '), 'success');
      } catch(e) { console.error('Auto GCal sync error', e); }
    }

    if (gcalIsConnected()) {
      // Token encore valide → sync direct
      await doSync();
    } else {
      // Token expiré ou absent → demande silencieuse (sans popup)
      if (!gcalTokenClient) gcalInit();
      if (gcalTokenClient) {
        try {
          gcalTokenClient.requestAccessToken({ prompt: '' });
          // Le callback dans gcalInit stocke le token et lance le sync auto
          // On n'a rien d'autre à faire ici
        } catch(e) { /* ignore si l'utilisateur n'a pas encore autorisé */ }
      }
    }
  }, 1500);
}
